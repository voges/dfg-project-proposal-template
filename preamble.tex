\documentclass[a4paper, 11pt]{article}

% Babel
\usepackage[english, ngerman]{babel}

% Personal data
\newcommand{\Title}{Title}
\newcommand{\Author}{Firstname Lastname}
\newcommand{\City}{City}

% Custom colors
\usepackage{xcolor}
\definecolor{azure}{HTML}{008aff}

% Page margins
\usepackage[
    top=2.5cm,
    bottom=1.5cm,
    left=2.5cm,
    right=2cm,
    includefoot
]{geometry}

% Font setup
\usepackage{fontspec}
\setsansfont{Arial}
\renewcommand{\familydefault}{\sfdefault}

% Paragraph indentation/spacing (zero \parindent and non-zero \parskip)
\usepackage{parskip}

% Define a new (numbered) \subsubsubsection{} command
\setcounter{secnumdepth}{4}
\makeatletter
\newcommand\subsubsubsection{\@startsection{paragraph}{4}{\z@}%
    {-2.5ex \@plus -1ex \@minus -.25ex}%
    {1.25ex \@plus .25ex}%
    {\normalfont\normalsize\bfseries}%
}
\makeatother

% Bibliography setup
\usepackage[
    giveninits=true,
    sorting=nyt,
    style=alphabetic
]{biblatex}
\addbibresource{references.bib}

% Highlighting of names in the bibliography based on item annotation
\newcommand{\highlightname}[1]{%
    \ifitemannotation{highlight}%
        {\textbf{#1}}%
        {#1}%
}
\AtEveryBibitem{
    \renewcommand*{\mkbibnamegiven}[1]{\highlightname{#1}}
    \renewcommand*{\mkbibnamefamily}[1]{\highlightname{#1}}
}

% Smaller bibliography font size
\renewcommand*{\bibfont}{\normalfont\footnotesize}

% Other packages
\usepackage{csquotes}

% Hyperlink setup (should be loaded last)
\usepackage{hyperref}
\hypersetup{
    pdftitle={\Title},
    pdfauthor={\Author},
    colorlinks=true,
    allcolors=azure
}
